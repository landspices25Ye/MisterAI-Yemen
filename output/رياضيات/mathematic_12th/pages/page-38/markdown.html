ولكن $‫و = هـ + ٢ ك‬$ ، حيث $‫ك = ص‬$‫

‬$‫\frac{هـ + ٢ ك}{و} = هـ‬$‫

‬$‫\therefore ع = \frac{١}{\sqrt{2}} [جنا \frac{هـ + ٢ ك}{و} + ت جا \frac{هـ + ٢ ك}{و} = \frac{١}{\sqrt{2}}]‬$‫

وبالتعويض عن قيم ‬$ك = ٠، ١، ٢، ...، و -١ نحصل على و من الجذور .

فمثلاً: إذا أردنا إيجاد الجذرين التربيعين للعدد ع نضع $‫ك = ٠، ١‬$ والجذور التكعيبية

نضع $‫ك = ٠، ١، ٢‬$ وهكذا .

مثال (١ - ٢٦)

أوجد الجذرين التربيعين للعدد ( $‫٢ - ٢ \sqrt{3}‬$ ت ) بالصيغتين القطبية والجبرية .

الحل :

أولاً : الصيغة القطبية :

نحول $‫ع = ٢ - ٢ \sqrt{3}‬$ ت إلى الصورة القطبية .

$‫س = ٢، ص = ٢ \sqrt{3} \Leftarrow \sqrt{س^2 + ص^2} = ٤ \sqrt{١٢ + ٤} = ١٦ \sqrt{١٦} = ٤‬$‫

‬$‫\therefore جنا = \frac{س}{٢} = \frac{١}{٢} \text{ ، جا = } \frac{ص}{٢} = \frac{\sqrt{3}}{2}‬$‫

‬$هـ = -٦٠$‫ ، لأنها تقع في الربع الرابع حيث جنا ‬$هـ > ٠$‫ ، جا ‬$هـ > ٠$‫ .

‬$‫\therefore هـ = ٣٦٠ - ٦٠ = ٣٠٠‬$‫

‬$‫ع = [س، هـ] = [٤، ٣٠٠]‬$

$‫ع = \sqrt{س} (جنا \frac{هـ + ٢ ك}{٢} + ت جا \frac{هـ + ٢ ك}{٢}) \text{ ، } ك = ٠، ١‬$‫

عندما ‬$ك = ٠$‫ ، فإن الجذر الأول = ‬$\sqrt{٤}$‫ (جنا ‬$\frac{٣٠٠}{٢} + ت جا \frac{٣٠٠}{٢}$‫) = ٢ (جنا ‬$١٥٠ + ت جا ١٥٠$‫ )

‬$‫= ٢ (\frac{\sqrt{3}}{2} + \frac{١}{2} ت) = -\sqrt{3} + ت‬$‫

عندما ‬$ك = ١$‫ ، فإن الجذر الثاني = ٢ (جنا ‬$٣٣٠ + ت جا ٣٣٠$‫ ) = ٢ ( ‬$\frac{\sqrt{3}}{2} - \frac{١}{2} ت) = -\sqrt{3} - ت$$
