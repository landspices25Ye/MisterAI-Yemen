التكامل

$‫\text{د (س}^* \text{) = د (١ + ٢/٣ م) = (١ + ٢/٣ م) + (١ + ٢/٣ م) + ١}‬$‫

‬$‫= ١ + \frac{٤}{٣} م + \frac{٤}{٣} م + ٢ + \frac{٢}{٣} م‬$‫

‬$‫= \frac{٤}{٣} م + ٢ + \frac{٦}{٣} م + ٣‬$‫

‬$‫\text{مَجْد ١/٣} \text{ د (س}^* \text{) } \Delta \text{ س} = \text{مَجْد ١/٣} \left( \frac{٤}{٣} م + ٢ + \frac{٦}{٣} م + ٣ \right) \times \frac{٢}{٣}‬$‫

‬$‫= \frac{٢}{٣} \left[ \frac{٢٤}{٣} \times \frac{١ (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (\text{مَجْد ١/٣}) \right]‬$‫

‬$‫= \frac{٢}{٣} \left[ \frac{١ (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + ٢) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (١ + 2) (\text{مَجْد ١/٣}) \right]‬$‫

‬$‫= \frac{٢}{٣} \left[ \frac{٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + 2}{\text{مَجْد ١/٣}) \right]‬$‫

‬$‫= \frac{٢}{٣} \left[ \frac{٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + 2}{\text{مَجْد ١/٣}) \right]‬$‫

‬$‫\therefore \text{سط } \frac{٣}{١} = \text{نها } \frac{٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + ٢ + 2}{\text{مَجْد ١/٣}) \text{ وحدة مربعة . [استناداً إلى الصيغة (١ - ٧) .]}}‬$‫

# تعريف (١ - ٧)

إذا كانت د دالة معرفة على الفترة [١، ب]، وكانت النهاية التالية موجودة.

‬$‫\text{نها } \text{مَجْد ١/٣} \text{ د (س}^* \text{) } \Delta \text{ س}‬$‫

عندئذ تسمى هذه النهاية التكامل المحدد للدالة د على الفترة [١، ب]

ويعبر عنها بالرمز ‬$\text{أ} \text{د (س) و س}$‫ . أي أن:

‬$‫\text{أ} \text{د (س) و س} = \text{نها } \text{مَجْد ١/٣} \text{ د (س}^* \text{) } \Delta \text{ س}‬$‫

‬$$‫\text{أ} \text{د (س) و س} = \text{نها } \text{مَجْد ١/٣} \text{ د (س}^* \text{) } \Delta \text{ س}‬$$‫

ملاحظات: - الصورة ‬$\text{أ} \text{د (س) و س}$‫ تقرأ تكامل د (س) بالنسبة لـ س، والرمز ‬$\text{أ} \text{د}$ علامة التكامل المحدد، ويسمى العدد (١) بالحد الأدنى، والعدد (ب) بالحد الأعلى لحدود التكامل.
