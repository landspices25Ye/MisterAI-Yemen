مثال (١ - ٢٨)

حل المعادلتين التاليتين :

أ) $‫ع^2 - ٢ع + ٥ = ٠‬$ .

ب) $‫ع^2 - (٥ + ٧)ت + (٦ - ١٧)ت = ٠‬$ .

الحل :

أ) $‫ع^2 - ٢ع + ٥ = ٠‬$ .

ب = ١ = ١ ، ب = ٢ - ٥ ، ج = ٥ .

$‫\Delta = ٢ - ١٤ج = ٤ - ٢٠ = ١٦‬$

$‫\therefore ع = \frac{\overline{\Delta} \sqrt{١٦ - ٢}}{٢} = \frac{\overline{\Delta} \sqrt{١٦ - ٢}}{٢} = ١ \pm ٢ت‬$ .

$‫ع_1 = ١ + ٢ت‬$ ، $‫ع_2 = ١ - ٢ت‬$ .

ب) $‫ع^2 - (٥ + ٧)ت + (٦ - ١٧)ت = ٠‬$ .

$‫١ = ١‬$ ، ب = $‫٢ - (٥ + ٧)ت‬$ ، ج = ٦ - ١٧ + ٦ت .

$‫\Delta = ٢ - ١٤ج = (٥ + ٧)ت^2 - ٤(٦ - ١٧)ت = ٢ت‬$ .

$‫\therefore ع = \frac{\overline{\Delta} \sqrt{١٦ - ٢}}{٢} = \frac{\overline{\Delta} \sqrt{١٦ - ٢}}{٢}‬$ .
