الأعداد المركبة

مثال (١ - ٢٤)

بسط ما يلي : $‫\frac{(\text{جتا هـ} - \text{ت جا هـ})^{10}}{(\text{جتا هـ} + \text{ت جا هـ})^{12}}‬$‫

الحل : البسط = [جتا هـ - ت جا هـ]‬$^{10}$‫ = [جتا (هـ) + ت جا (هـ)]‬$^{10}$‫

= جتا (١٠هـ) + ت جا (١٠هـ) (مبرهنة دي موافر)

المقام = [جتا هـ + ت جا هـ]‬$^{12}$‫ = جتا (١٢هـ) + ت جا (١٢هـ)

‬$‫\therefore \frac{\text{جتا (١٠هـ) + ت جا (١٠هـ)}}{\text{جتا (١٢هـ) + ت جا (١٢هـ)}} = \frac{[\text{١٠هـ}, 1]}{[\text{١٢هـ}, 1]} = [\frac{1}{1}, -10\text{هـ} - 12\text{هـ}] = [1, -22\text{هـ}]‬$‫

مثال (١ - ٢٥)

عبر عن جتا ٢ هـ ، جا ٢ هـ بدلالة جا هـ ، جتا هـ ؛ باستخدام مبرهنة دي موافر

الحل : (جتا هـ + ت جا هـ)‬$^{2}$‫ = جتا‬$^{2}$‫ هـ - جا‬$^{2}$‫ هـ + ٢ ت جا هـ جتا هـ

وحسب مبرهنة دي موافر (جتا هـ + ت جا هـ)‬$^{2}$‫ = (جتا ٢ هـ + ت جا ٢ هـ)

ومن تساوي عددين مركبين ينتج أن:

جتا ٢ هـ = جتا‬$^{2}$‫ هـ - جا‬$^{2}$‫ هـ

جا ٢ هـ = ٢ جا هـ جتا هـ

تدريب (١ - ٨)

أوجد جتا ٣ هـ ، جا ٣ هـ بدلالة جا هـ ، جتا هـ .

ثانياً : الجذور :

إذا كان و ط / { ١, ٠ } ، وكان ا ، ع م ، فإن المعادلة ع = ا يوجد لها و من الجذور.

نفرض أن ا = م (جتا هـ + ت جا هـ) ، ع = م (جتا هـ + ت جا هـ)

بالتعويض في المعادلة ع = ا

م‬$_{1}$‫ (جتا هـ + ت جا هـ)‬$^{2}$‫ = م (جتا هـ + ت جا هـ)

م‬$_{2}$‫ [جتا (و هـ) + ت جا (و هـ)] = م (جتا هـ + ت جا هـ) ... لماذا ؟
