مثال (٦ - ١٢)

أوجد مشتقة كل ما يأتي :

أ) ص² + س ص + س² = ٣ س ب) س ص² - س² ص = ٦

الحل :

أ) ٢ ص² + س ص² + س² = ٣ س ب) ٢ س ص² - س² ص = ٦

$‫\frac{ص^2 - س^2 - ٣}{س + س^2} = \frac{ص^2 - س^2 - ٣}{س + س^2}‬$‫

ب) ص² + ٢ س ص² - س² - س² = ٠

‬$‫\frac{ص^2 - س^2 - ٢}{س - س^2} = \frac{ص^2 - س^2 - ٢}{س - س^2}‬$‫

مثال (٦ - ١٣)

أوجد ‬$\frac{ص^2}{س}$‫ للدالة ‬$\sqrt{س + ص} = ١ - ٢$‫

الحل :

‬$‫\frac{ص^2}{س} = \frac{\frac{ص^2}{س} + ١}{\frac{ص^2}{س} + ٢}‬$‫

‬$‫\frac{ص^2}{س} - \frac{ص^2}{س} \quad \text{ص} \quad ١ = \frac{ص^2}{س} - ١‬$‫

‬$$‫\frac{ص^2}{س} - \frac{ص^2}{س} \quad \text{ص} \quad ١ = \frac{ص^2}{س} - ١‬$$‫

مثال (٦ - ١٤)

أوجد ‬$\frac{ص^2}{س}$‫ للدالة س ص² - ص = ٤

الحل :

س × ٢ ص² + س² + ١ × ١ = ١ ص² - ١ = ٢ س ص² - ١ ص² = ١ - ص²

‬$‫\frac{ص^2 - ١}{س - ١} = \frac{ص^2 - س^2 - ١}{س - ١}‬$
