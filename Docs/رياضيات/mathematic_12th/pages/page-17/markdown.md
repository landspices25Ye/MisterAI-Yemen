الأعداد المركبة

٤) النظير الجمعي للعدد ع = س + ت ص هو -ع = -س - ت ص .

أي أن : ع + (ع) = ٠ ، ٧ ع Ⓜ م

لإثبات ذلك :

نفرض أن النظير الجمعي للعدد (س ، ص) هو (سّ ، صّ) ،

(سّ ، صّ) + (س ، ص) = (٠ ، ٠) [ تعريف النظير ]

(سّ + س ، صّ + ص) = (٠ ، ٠) [ تساوي عددين مركبين ] .

سّ + س = ٠ ⇐ سّ = -س

صّ + ص = ٠ ⇐ صّ = -ص

∴ (سّ ، صّ) = (س - ، ص -)

∴ النظير الجمعي للعدد المركب ع هو -ع = -س - ت ص = - (س + ت ص) = -ع

# **مثال (١ - ٨)**

لتكن ع₁ = ٣ + ت ، ع₂ = ١/٢ - ٥ ت ، ع₃ = ٦ - ت ؛ أوجد ما يلي :

أ) ع₁ + ع₂ . ب) ع₃ - ع₄ . ج) ع₁ + ع₂ - ع₃ .

# **الحل :**

أ) ع₁ + ع₂ = (٣ + ت) + (١/٢ - ٥ ت)

= (٣ + ١/٢) + (١ - ٥) ت [ تعريف الجمع ]

= ١/٢ + ٣ + (٤ - ٢) ت = ٧/٢ - ٤ ت

ب) ع₃ - ع₄ = (٦ - ت) - (٣ + ت)

= (٦ - ٣) + (١ - ١) ت = ٣ + (٢ - ٢) ت = ٣ - ٢ ت [ تعريف الطرح ]

ج) ع₁ + ع₂ - ع₃ = (٣ + ت) + (٦ - ت) - (١/٢ - ٥ ت)

= (٣ + ٦) + (١ - ١) ت - (١/٢ - ٥ ت) [ تعريف الجمع ]

= (٩ + ٠) ت - (١/٢ - ٥ ت)

= (٩ - ١/٢) + (٠ + ٥) ت = ٨/٢ + ٥ ت [ تعريف الطرح ]
