# مثال (١ - ٣٠)

إذا كانت ع + ١/ع = ٢ جتا هـ ، ع ≠ ٠

فأثبت أن : ع² + ١/ع² = ٢ جتا و هـ .

الحل :

ع + ١/ع = ٢ جتا هـ ⇔ ع² + ١ = ٢ جتا هـ ⇔ ع² - ٢ جتا هـ + ١ = ٠

ع = ٢ جتا هـ ± ٢ √جتا² هـ - ١/٢

ع = جتا هـ ± √جتا² هـ

ع = جتا هـ ± ت جا هـ

بالتعويض في المعادلة التالية :

ع² + ١/ع² = (جتا هـ ± ت جا هـ)² + ١/(جتا هـ ± ت جا هـ)² .

= (جتا هـ ± ت جا هـ)² + (جتا هـ ± ت جا هـ)² =

= جتا (و هـ) ± ت جا (و هـ) + جتا (-و هـ) ± ت جا (-و هـ)

= [جتا (و هـ) ± ت جا (و هـ) + جتا (و هـ) ± ت جا (و هـ)]

= ٢ جتا (و هـ) .

# مثال (١ - ٣١)

كون المعادلة التي جذراها ٢ - ت ، ٢ + ٣ ت .

الحل :

المعادلة التي لها جذران هي معادلة من الدرجة الثانية .

مجموع الجذرين = (٢ - ت) + (٢ + ٣) = ٥ + ت

حاصل ضرب الجذرين = (٢ - ت) + (٢ + ٣) = ٨ + ت

∴ المعادلة هي :

ع² - (مجموع الجذرين) ع + (حاصل ضرب الجذرين) = ٠ .

ع² - (٥ + ت) ع + (٨ + ت) = ٠ .
